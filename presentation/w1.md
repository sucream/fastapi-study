# 1. DB connection

```python
# 메모리로만 사용
DB_URL = "sqlite:///:memory:"
engine = create_engine(DB_URL, connect_args={"check_same_thread": False})
SessionLocal = sessionmaker(autocommit=False, autoflush=False, bind=engine)

def get_db():
    """
    DB 세션을 생성합니다.
    """

    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()
```
